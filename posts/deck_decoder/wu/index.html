def calculate_entropy(data):
    if not data:
        return 0
    frequency = Counter(data)
    data_len = len(data)
    probabilities = [freq / data_len for freq in frequency.values()]
    entropy = -sum(prob * math.log2(prob) for prob in probabilities if prob > 0)
    return entropy
